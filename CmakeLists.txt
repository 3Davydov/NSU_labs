cmake_minimum_required(VERSION 3.8)

set(This BigInt)

project(${This} C CXX)

set(CMAKE_C_STANDARD 99)
set(CMAKE_CXX_STANDARD 11)
set(CMAKE_POSITION_INDEPENDENT_CODE ON)

enable_testing()

add_subdirectory(googletest)
set(Header
    BigInt.hpp
)
set(Sources
    BigInt_Class_members.cpp
    BigInt_Supportive_Functions.cpp
    BigInt_main.cpp
)

add_executable(BigInt_m BigInt_main.cpp BigInt_Class_members.cpp  BigInt_Supportive_Functions.cpp BigInt.hpp)

add_library(${This} STATIC BigInt_Class_members.cpp  BigInt_Supportive_Functions.cpp BigInt.hpp)

add_subdirectory(Tests)
